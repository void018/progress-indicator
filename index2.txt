    <script>
        // PASTE YOUR GOOGLE SCRIPT URL INSIDE THE QUOTES BELOW
        const DATA_URL = "YOUR_GOOGLE_SCRIPT_URL_HERE";

        async function initTracker() {
            try {
                const response = await fetch(DATA_URL);
                const data = await response.json();

                // 1. Update Text
                document.getElementById('current-display').innerText = formatCurrency(data.current);
                document.getElementById('target-display').innerText = formatCurrency(data.target);
                document.getElementById('loading-msg').style.display = 'none';

                // 2. Calculate Percentage
                let percentage = (data.current / data.target) * 100;
                if (percentage > 100) percentage = 100; // Cap at 100%
                
                // 3. Update Bar Width
                document.getElementById('progress-bar').style.width = percentage + "%";

                // 4. Create Milestones
                const track = document.getElementById('progress-track');
                
                data.milestones.forEach(percentVal => {
                    // Safety check to ensure it fits on bar
                    if(percentVal >= 0 && percentVal <= 100) {
                        const marker = document.createElement('div');
                        marker.classList.add('milestone-marker');
                        marker.style.left = percentVal + "%";
                        marker.title = "Milestone at " + percentVal + "%";
                        track.appendChild(marker);
                    }
                });

            } catch (error) {
                console.error("Error fetching data:", error);
                document.getElementById('loading-msg').innerText = "Error loading data.";
            }
        }

        // Helper to format money (e.g. 1000 -> $1,000)
        function formatCurrency(num) {
            return "$" + Number(num).toLocaleString();
        }

        // Run on load
        window.addEventListener('DOMContentLoaded', initTracker);
    </script>
